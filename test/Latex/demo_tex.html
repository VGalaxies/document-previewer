public class Demo {
    public static int binaryOne(int n) {
        if (n <= 0)
            throw new RuntimeException();

        if (n == 1) {
            return 1;
        } else if (n % 2 == 0) {
            return binaryOne(n / 2);
        } else {
            return 1 + binaryOne(n / 2);
        }
    }


    public static void main(String[] args) {
        binaryOne(3);
    }
}
